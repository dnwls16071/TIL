```python
def convert(n):
    zero_cnt = 0
    conv_cnt = 0
    while len(n) != 1:
        result = ""
        for i in n:
            if int(i) == 0:
                zero_cnt += 1
            else:
                result += "1"
        len_res = len(result)
        answer = []
        while len_res != 0:
            answer.append(len_res % 2)
            len_res //= 2
        answer = answer[::-1]
        n = answer
        conv_cnt += 1
    return [conv_cnt, zero_cnt]
```

```python
def solution(s):
    change, zero = 0, 0
    while s != '1':
        change += 1
        num = s.count('1')
        zero += len(s) - num
        s = bin(num)[2:]
    return [change, zero]
```
